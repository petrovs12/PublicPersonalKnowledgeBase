
# 2D

$\mathbf{R}(\theta) = \left[\begin{array}
{rrr}
cos(\theta) & -sin(\theta)  \\
sin(\theta) & cos(\theta)  \\
\end{array}\right]
$

$\mathbf{R}(\pi/2) = \left[\begin{array}
{rrr}
0 & -1  \\
1 & 0  \\
\end{array}\right]
$

$\mathbf{R}(\pi) = \left[\begin{array}
{rrr}
1 & 0  \\
0 & -1  \\
\end{array}\right]
$

$\mathbf{R}(3\pi/2) = \left[\begin{array}
{rrr}
0 & 1  \\
-1 & 0  \\
\end{array}\right]
$
